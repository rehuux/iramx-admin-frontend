async function login() { const username = document.getElementById("username").value.trim(); const password = document.getElementById("password").value.trim(); const res = await fetch(`${ADMIN_BASE}/api/admin/login`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username, password }) }); const data = await res.json(); if (data.success) { localStorage.setItem("admin-token", "REHU_ADMIN_TOKEN_786"); window.location.href = "admin-dashboard.html"; } else { alert("Invalid Login"); } }
