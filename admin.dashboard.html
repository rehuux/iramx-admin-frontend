async function save() { const token = localStorage.getItem("admin-token"); const body = { aadhaar_base: document.getElementById("aadhaarBase").value, aadhaar_key: document.getElementById("aadhaarKey").value, mobile_in_base: document.getElementById("mobileBase").value, mobile_in_key: document.getElementById("mobileKey").value, banner_text: document.getElementById("bannerText").value }; const res = await fetch(`${ADMIN_BASE}/api/admin/saveSettings`, { method: "POST", headers: { "Content-Type": "application/json", "Authorization": token }, body: JSON.stringify(body) }); const data = await res.json(); if (data.success) { alert("Settings saved!"); } else { alert("Error: " + data.error); } }
